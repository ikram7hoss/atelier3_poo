#include <iostream>
using namespace std;

class Fichier {
private:
    char *P;
    int longueur;
    
public:
    Fichier() {
        P = NULL;
        longueur = 0;
        cout << "Constructeur appelé" << endl;
    }
    
    void Creation(int taille) {
        longueur = taille;
        P = new char[longueur];
        cout << "Fichier créé avec " << longueur << " octets" << endl;
    }
    
    void Remplit() {
        for (int i = 0; i < longueur; i++) {
            P[i] = 'A' + (i % 26);  // remplit avec A, B, C...
        }
        cout << "Fichier rempli" << endl;
    }
    
    void Affiche() {
        cout << "Contenu du fichier : ";
        for (int i = 0; i < longueur; i++) {
            cout << P[i];
        }
        cout << endl;
    }
    
    ~Fichier() {
        if (P != NULL) {
            delete[] P;
        }
        cout << "Destructeur appelé" << endl;
    }
};

int main() {
    Fichier *f;
    
    f = new Fichier();
    
    f->Creation(20);
    f->Remplit();
    f->Affiche();
    
    delete f;
    
    return 0;
}
