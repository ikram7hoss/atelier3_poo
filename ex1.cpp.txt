#include <iostream>
using namespace std;

class Pile {
private:
    int tab[100];
    int sommet;
    
public:
    Pile() {
        sommet = 0;  // petit oubli : devrait être -1 mais ça marche quand même
    }
    
    void push(int val) {
        if (sommet < 100) {
            tab[sommet] = val;
            sommet++;
            cout << "Empilé : " << val << endl;
        }
        else {
            cout << "Pile pleine" << endl;
        }
    }
    
    int pop() {
        if (sommet > 0) {
            sommet--;
            int val = tab[sommet];
            return val;
        }
        else {
            cout << "Pile vide" << endl;
            return -1;
        }
    }
};

int main() {
    Pile p1, p2;
    
    cout << "=== Test pile p1 ===" << endl;
    p1.push(10);
    p1.push(20);
    p1.push(30);
    
    cout << "Dépilé : " << p1.pop() << endl;
    cout << "Dépilé : " << p1.pop() << endl;
    
    cout << "\n=== Test pile p2 ===" << endl;
    p2.push(5);
    p2.push(15);
    p2.push(25);
    p2.push(35);
    
    cout << "Dépilé : " << p2.pop() << endl;
    cout << "Dépilé : " << p2.pop() << endl;
    cout << "Dépilé : " << p2.pop() << endl;
    
    return 0;
}
